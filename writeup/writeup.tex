\newcommand{\chessboard}{
    \foreach \x in {0,1,...,\numexpr\boardsize-1}{
        \foreach \y in {0,1,...,\numexpr\boardsize-1}{
            \pgfmathtruncatemacro{\shade}{mod(\x+\y,2)}
            \ifnum\shade=0
                \fill[darksq] (\x, \y) rectangle ++(1,1);
            \else
                \fill[lightsq] (\x, \y) rectangle ++(1,1);
            \fi
        }
    }
    % Draw the hole
    \fill[white] ({(\boardsize - \holesize)/2}, {(\boardsize - \holesize)/2}) rectangle ++(\holesize, \holesize);

    % cut corners
    \fill[white] (-0.01,-0.01) -- (-0.01,\cornercut) -- (\cornercut,-0.01) -- cycle;
    \fill[white] (\boardsize+0.01,\boardsize+0.01) -- (\boardsize+0.01,\boardsize-\cornercut) -- (\boardsize-\cornercut,\boardsize+0.01) -- cycle;
    \fill[white] (-0.01,\boardsize) -- (-0.01,\boardsize-\cornercut) -- (\cornercut,\boardsize+0.01) -- cycle;
    \fill[white] (\boardsize,-0.01) -- (\boardsize+0.01,\cornercut) -- (\boardsize-\cornercut,-0.01) -- cycle;
}
